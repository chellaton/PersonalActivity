  summarize(AverageSteps=mean(steps, na.rm=TRUE))

wkendPlot <- ggplot(filter(PAdata_summ,Weekday==0),aes(x=interval, y=AverageSteps))+
                      geom_line(col="dark red")+ylab("Weekend Average Steps")
